def spawn(*args, &block)

  if block_given?
    ::PTY.spawn(*args) do |reader, writer, pid|
      begin
        yield(reader, writer, pid)
      rescue Errno::EIO
      ensure
        ::Process.wait(pid)
      end
    end
  else
    reader, writer, pid = ::PTY.spawn(*args)
  end

  [reader, writer, pid]
end
