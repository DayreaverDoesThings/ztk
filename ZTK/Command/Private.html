def log_header(what)
  count = 8
  sep = ("=" * count)
  header = [sep, "[ #{what} ]", sep, "[ #{tag} ]", sep, "[ #{what} ]", sep].join
  "#{header}\n"
end

def tag
  @@hostname ||= Socket.gethostname.split('.').first.strip
  "#{ENV['USER']}@#{@@hostname}"
end
