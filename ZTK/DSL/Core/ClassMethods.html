def cattr_reader(*args)
  args.flatten.each do |arg|
    next if arg.is_a?(Hash)
    instance_eval %Q{
      unless defined?(@@#{arg})
        @@#{arg} = nil
      end

      def #{arg}
        @@#{arg}
      end
    }
  end
end

def cattr_writer(*args)
  args.flatten.each do |arg|
    next if arg.is_a?(Hash)
    instance_eval %Q{
      unless defined?(@@#{arg})
        @@#{arg} = nil
      end

      def #{arg}=(value)
        @@#{arg} = value
      end
    }
  end
end
